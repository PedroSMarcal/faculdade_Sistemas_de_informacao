CREATE TABLE mulher (
id_mulher		INTEGER,
nome_mulher		VARCHAR2(15),
CONSTRAINT pk_tb_mulher PRIMARY KEY(id_mulher)
);

CREATE TABLE homem (
id_homem    INTEGER,
nome_homem    VARCHAR2(15),
id_mulher   INTEGER,
CONSTRAINT pk_tb_homem PRIMARY KEY (id_homem),
CONSTRAINT fk_tb_homem_tb_mulher FOREIGN KEY (id_mulher) REFERENCES mulher(id_mulher)
);

b) 
SELECT h.nome_homem || ' Casado com a ' || m.nome_mulher as "Casamento" FROM homem h NATURAL JOIN mulher m;

c)
SELECT h.nome_homem || ' Casado com a ' || m.nome_mulher as "Casamento" FROM homem h JOIN mulher m USING (id_mulher);

d)
SELECT h.nome_homem || ' Casado com a ' || m.nome_mulher as "Casamento" FROM homem h JOIN mulher m ON h.id_mulher = m.id_mulher;

e)
