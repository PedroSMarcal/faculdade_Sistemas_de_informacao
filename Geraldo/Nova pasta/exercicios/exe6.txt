CREATE TABLE mulher (
id_mulher		INTEGER,
nome_mulher		VARCHAR2(15),
CONSTRAINT pk_tb_mulher PRIMARY KEY(id_mulher)
);

CREATE TABLE homem (
id_homem    INTEGER,
nome_homem    VARCHAR2(15),
id_mulher   INTEGER,
CONSTRAINT pk_tb_homem PRIMARY KEY (id_homem),
CONSTRAINT fk_tb_homem_tb_mulher FOREIGN KEY (id_mulher) REFERENCES mulher(id_mulher)
);
