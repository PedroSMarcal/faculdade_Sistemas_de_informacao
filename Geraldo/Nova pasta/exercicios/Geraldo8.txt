SELECT
  CASE
    WHEN LENGTH(nome || ' ' || sobrenome) <= 10 THEN 
    nome || ' ' || sobrenome
    WHEN LENGTH(nome || ' ' || sobrenome) > 10 THEN 
    substr(nome, 1, 1) || ' ' || substr(sobrenome, 1, 10) 
  END 
FROM tb_empregado;

SELECT TO_CHAR(hf.data_termino, 'yyyy') as "Ano de Saída", COUNT(1) as "Quantidade", hf.id_funcao as "ID Função"
FROM tb_empregado e
INNER JOIN tb_historico_funcao hf ON e.id_empregado = hf.id_empregado
GROUP BY TO_CHAR(hf.data_termino, 'yyyy'), hf.id_funcao
ORDER BY COUNT(1) ASC;
