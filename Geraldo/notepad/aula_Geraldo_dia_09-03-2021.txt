--ALUNOS
DROP TABLE "tb_aluno";

CREATE TABLE "tb_aluno"(
cod_aluno	INTEGER,
nm_aluno	VARCHAR(40),
cod_curso	INTEGER,
CONSTRAINT pk_tb_aluno_cod_aluno PRIMARY KEY(cod_aluno),
CONSTRAINT fk_tb_aluno_cod_curso FOREIGN KEY(cod_curso)
	REFERENCES tb_curso(cod_curso)
);
